% - - - Text Formatting - - - %
\newcommand{\EG}[1]{
	\ifComments%
		\textcolor{blue}{(\textbf{EG: #1})}%
	\fi%
	}

\newcommand*{\code}[1]{\texttt{#1}}

\newcommand{\gfunc}[4]{
	\noindent
	\texttt{#1}(\textit{#2}):\hfill#3%
	\begin{adjustwidth}{0.75cm}{0cm}
		\small #4
	\end{adjustwidth}
	\medskip
	}

\newcommand{\func}[3]{\gfunc{#1}{#2}{}{#3}}

\newcommand{\pfunc}[3]{\gfunc{#1}{#2}{Private Method}{#3}}
	
\newcommand{\vari}[3]{
	\noindent
	\texttt{#1}\hfill#2
	\begin{adjustwidth}{0.75cm}{0cm}
		\small #3
	\end{adjustwidth}
	\medskip
	}

\newcommand{\remark}[1]{\textbf{Remark}:\ #1}

\newcommand{\keyword}[1]{= \textnormal{\code{#1}}}
